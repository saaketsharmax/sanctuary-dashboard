### 8.2 Database Schema (Supabase)

```sql
-- =============================================
-- CORE ENTITIES
-- =============================================

-- Users (extends Supabase auth.users)
CREATE TABLE public.users (
  id UUID PRIMARY KEY REFERENCES auth.users(id),
  email TEXT UNIQUE NOT NULL,
  name TEXT NOT NULL,
  avatar_url TEXT,
  user_type TEXT CHECK (user_type IN ('founder', 'partner', 'mentor', 'resident')),
  partner_sub_type TEXT CHECK (partner_sub_type IN ('vc', 'startup_manager')),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Founders
CREATE TABLE public.founders (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  startup_id UUID REFERENCES startups(id),
  bio TEXT,
  linkedin_url TEXT,
  role TEXT,
  previous_startups TEXT[],
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Startups
CREATE TABLE public.startups (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT NOT NULL,
  one_liner TEXT,
  stage TEXT,
  industry TEXT,
  business_model TEXT,
  status TEXT DEFAULT 'active',
  cohort TEXT,
  accepted_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Mentors
CREATE TABLE public.mentors (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  bio TEXT,
  expertise TEXT[],
  industries TEXT[],
  availability JSONB,
  linkedin_url TEXT,
  rating DECIMAL(3,2),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- =============================================
-- APPLICATION & ONBOARDING
-- =============================================

-- Applications
CREATE TABLE public.applications (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id),
  status TEXT DEFAULT 'draft',
  form_data JSONB NOT NULL,
  submitted_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Interviews
CREATE TABLE public.interviews (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  application_id UUID REFERENCES applications(id),
  transcript JSONB[],
  signals JSONB[],
  scores JSONB,
  status TEXT DEFAULT 'pending',
  started_at TIMESTAMPTZ,
  completed_at TIMESTAMPTZ
);

-- Research Results
CREATE TABLE public.research_results (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  application_id UUID REFERENCES applications(id),
  market_analysis JSONB,
  competitor_analysis JSONB,
  timing_analysis JSONB,
  founder_background JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Decisions
CREATE TABLE public.decisions (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  application_id UUID REFERENCES applications(id),
  status TEXT CHECK (status IN ('accepted', 'waitlisted', 'declined')),
  decided_by UUID REFERENCES users(id),
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- =============================================
-- STARTUP MANAGEMENT
-- =============================================

-- Metrics
CREATE TABLE public.metrics (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  startup_id UUID REFERENCES startups(id),
  period DATE NOT NULL,
  mrr DECIMAL(12,2),
  arr DECIMAL(12,2),
  total_users INTEGER,
  active_users INTEGER,
  paying_customers INTEGER,
  growth_rate DECIMAL(5,2),
  burn_rate DECIMAL(12,2),
  runway_months INTEGER,
  cac DECIMAL(10,2),
  ltv DECIMAL(10,2),
  churn_rate DECIMAL(5,2),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Documents
CREATE TABLE public.documents (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  startup_id UUID REFERENCES startups(id),
  name TEXT NOT NULL,
  type TEXT,
  url TEXT NOT NULL,
  uploaded_by UUID REFERENCES users(id),
  shared_with UUID[],
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Checkpoints
CREATE TABLE public.checkpoints (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  startup_id UUID REFERENCES startups(id),
  title TEXT NOT NULL,
  description TEXT,
  target_date DATE,
  status TEXT DEFAULT 'pending',
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  completed_at TIMESTAMPTZ
);

-- Mentor Matches
CREATE TABLE public.mentor_matches (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  startup_id UUID REFERENCES startups(id),
  mentor_id UUID REFERENCES mentors(id),
  blocker TEXT,
  status TEXT DEFAULT 'pending',
  session_date TIMESTAMPTZ,
  session_notes TEXT,
  feedback_startup JSONB,
  feedback_mentor JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- =============================================
-- COMMUNITY HUB
-- =============================================

-- KYR Profiles
CREATE TABLE public.kyr_profiles (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  preferred_name TEXT,
  pronouns TEXT,
  role TEXT,
  company TEXT,
  working_on TEXT,
  expertise TEXT[],
  seeking TEXT[],
  work_style TEXT,
  coffee_order TEXT,
  superpower TEXT,
  hobbies TEXT[],
  linkedin_url TEXT,
  is_public BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Events
CREATE TABLE public.events (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  title TEXT NOT NULL,
  description TEXT,
  category TEXT,
  date DATE NOT NULL,
  start_time TIME,
  end_time TIME,
  location TEXT,
  is_virtual BOOLEAN DEFAULT false,
  virtual_link TEXT,
  capacity INTEGER,
  host_id UUID REFERENCES users(id),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- RSVPs
CREATE TABLE public.rsvps (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  event_id UUID REFERENCES events(id) ON DELETE CASCADE,
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  status TEXT CHECK (status IN ('attending', 'maybe', 'declined')),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(event_id, user_id)
);

-- Check-ins
CREATE TABLE public.check_ins (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id),
  location TEXT,
  status_message TEXT,
  check_in_time TIMESTAMPTZ DEFAULT NOW(),
  check_out_time TIMESTAMPTZ
);

-- Board Posts
CREATE TABLE public.board_posts (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  author_id UUID REFERENCES users(id),
  title TEXT NOT NULL,
  content TEXT NOT NULL,
  category TEXT,
  reactions JSONB DEFAULT '{}',
  is_pinned BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Visitors
CREATE TABLE public.visitors (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT NOT NULL,
  company TEXT,
  purpose TEXT,
  host_id UUID REFERENCES users(id),
  badge_number TEXT,
  check_in_time TIMESTAMPTZ DEFAULT NOW(),
  check_out_time TIMESTAMPTZ
);

-- Celebrations
CREATE TABLE public.celebrations (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id),
  type TEXT CHECK (type IN ('birthday', 'work_anniversary', 'funding', 'launch', 'milestone')),
  title TEXT NOT NULL,
  description TEXT,
  date DATE NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- INDEXES

CREATE INDEX idx_applications_user ON applications(user_id);

CREATE INDEX idx_metrics_startup ON metrics(startup_id);
CREATE INDEX idx_metrics_period ON metrics(period);
CREATE INDEX idx_events_date ON events(date);
CREATE INDEX idx_check_ins_user ON check_ins(user_id);
CREATE INDEX idx_check_ins_time ON check_ins(check_in_time);
```
