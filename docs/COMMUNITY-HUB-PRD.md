## Data Model

### Core Tables

```sql
-- Residents (extends users)
CREATE TABLE residents (
  id UUID PRIMARY KEY REFERENCES users(id),

  -- Basic Info
  pronouns TEXT,
  birthday DATE,
  languages TEXT[],
  bio TEXT,
  fun_facts TEXT[],

  -- Work
  skills TEXT[],
  can_help_with TEXT[],
  looking_for_help TEXT[],

  -- Preferences
  work_style JSONB, -- {earlyBird: bool, focusTimePrefs: [], commStyle: []}
  food_preferences JSONB, -- {dietary: [], allergies: [], cuisines: [], coffeeOrder: ''}

  -- Sports & Recreation
  sports JSONB[], -- [{sport: 'basketball', level: 'intermediate', lookingForPartner: true}]
  hobbies TEXT[],

  -- Social
  interests JSONB, -- {books: [], music: [], movies: [], games: []}

  -- Status
  is_active BOOLEAN DEFAULT true,
  joined_at TIMESTAMPTZ DEFAULT NOW(),

  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Attendance / Check-ins
CREATE TABLE attendance (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  resident_id UUID REFERENCES residents(id),
  checked_in_at TIMESTAMPTZ DEFAULT NOW(),
  checked_out_at TIMESTAMPTZ,
  location TEXT DEFAULT 'main', -- for multi-location support

  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Events
CREATE TABLE events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  description TEXT,
  event_type TEXT NOT NULL, -- workshop, fitness, sports, social, speaker, etc.

  start_time TIMESTAMPTZ NOT NULL,
  end_time TIMESTAMPTZ NOT NULL,
  location TEXT,

  capacity INTEGER,
  resident_reserved_capacity INTEGER, -- 70% rule

  created_by UUID REFERENCES users(id),

  -- Recurring
  is_recurring BOOLEAN DEFAULT false,
  recurrence_rule TEXT, -- RRULE format

  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Event Registrations
CREATE TABLE event_registrations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  event_id UUID REFERENCES events(id),
  resident_id UUID REFERENCES residents(id),

  status TEXT DEFAULT 'registered', -- registered, waitlisted, cancelled, attended
  registered_at TIMESTAMPTZ DEFAULT NOW(),

  UNIQUE(event_id, resident_id)
);

-- Visitors
CREATE TABLE visitors (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  email TEXT,
  company TEXT,

  visiting_resident_id UUID REFERENCES residents(id),
  purpose TEXT,

  visit_date DATE NOT NULL,
  expected_arrival TIME,
  checked_in_at TIMESTAMPTZ,
  checked_out_at TIMESTAMPTZ,

  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Community Board Posts
CREATE TABLE board_posts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  resident_id UUID REFERENCES residents(id),

  post_type TEXT NOT NULL, -- looking_for, offering, lost_found, general
  content TEXT NOT NULL,

  is_resolved BOOLEAN DEFAULT false,
  expires_at TIMESTAMPTZ,

  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Shoutouts
CREATE TABLE shoutouts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  from_resident_id UUID REFERENCES residents(id),
  to_resident_id UUID REFERENCES residents(id),

  message TEXT NOT NULL,

  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Coffee Roulette Matches
CREATE TABLE coffee_matches (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  resident_1_id UUID REFERENCES residents(id),
  resident_2_id UUID REFERENCES residents(id),

  week_of DATE NOT NULL,
  status TEXT DEFAULT 'pending', -- pending, scheduled, completed, skipped

  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Sports Games
CREATE TABLE sports_games (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  sport TEXT NOT NULL,

  scheduled_at TIMESTAMPTZ NOT NULL,
  location TEXT,

  min_players INTEGER,
  max_players INTEGER,
  skill_level TEXT, -- any, beginner, intermediate, advanced

  organizer_id UUID REFERENCES residents(id),

  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Sports Game Participants
CREATE TABLE sports_participants (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  game_id UUID REFERENCES sports_games(id),
  resident_id UUID REFERENCES residents(id),

  status TEXT DEFAULT 'joined', -- joined, maybe, cancelled

  UNIQUE(game_id, resident_id)
);

-- News/Announcements
CREATE TABLE announcements (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  title TEXT NOT NULL,
  content TEXT NOT NULL,
  announcement_type TEXT, -- achievement, welcome, general, milestone

  related_startup_id UUID,
  related_resident_id UUID REFERENCES residents(id),

  is_pinned BOOLEAN DEFAULT false,
  published_at TIMESTAMPTZ DEFAULT NOW(),

  created_by UUID REFERENCES users(id),
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```
